```typescript
interface RetrievedSource {
  retrievalId: string;            // "src_1"..."src_N" in ranked order
  chunkRef: string;               // "{corpus}:{id}"
  corpus: "pubmed" | "youtube" | "podcast" | "industry" | "manufacturer" | "fda" | "practice";
  scores: { vector?: number; fts?: number; fused: number; reranked?: number; authorityWeight: number; final: number };
  text: string;                   // chunk/snippet passed to the LLM
  locator: SourceLocator;
}
interface PubMedLocator { type: "pubmed"; pmid: string; title: string; authors?: string; journal?: string; pubDate?: string; doi?: string; url: string; }
interface DocumentLocator { type: "document"; corpus: "manufacturer" | "fda" | "practice"; filename: string; title: string; pageNumber?: number; section?: string; storagePath: string; url: string; sourceAuthority?: "fda_label" | "manufacturer" | "clinical_study" | "internal"; }
type ResearchEvent =
  | { type: "status"; stage: "searching" | "ranking" | "generating" }
  | { type: "sources"; sources: RetrievedSource[] }
  | { type: "token"; text: string }
  | { type: "citations"; citations: ResearchCitation[]; displayMap: Record<string, number> }
  | { type: "done"; runId: string; usage: { inputTokens: number; outputTokens: number; durationMs: number } }
  | { type: "error"; code: string; message: string; retryable: boolean };
```

    Header: `import { NextRequest } from "next/server";` + `export const dynamic = "force-dynamic";`. Import `retrieveSources` from `@/lib/retrieval/sources`, `resolveCitations` from `@/lib/retrieval/post-process`, `gateway` from `@ai-sdk/gateway`, `streamText` from `ai`, types from `@/lib/types/retrieval`.

    `export async function POST(req: NextRequest)`:
    - `const { query } = await req.json();` — if `!query`, return a 400 JSON `{ error: "query required" }`.
    - Build a `ReadableStream` with an `emit(event: ResearchEvent)` helper that does `controller.enqueue(encoder.encode(\`data: ${JSON.stringify(event)}\n\n\`))`.
    - Sequence inside `start(controller)`:
      1. `emit({ type: "status", stage: "searching" })`
      2. `const sources = await retrieveSources(query);`
      3. `emit({ type: "status", stage: "ranking" })`
      4. `emit({ type: "sources", sources })`
      5. If `sources.length === 0` (D-08 graceful no-results): emit a single token event with text `"No grounded sources were found for this question in the current evidence library."`, then `emit citations [] / displayMap {}`, then `done`, `controller.close()`, return.
      6. `emit({ type: "status", stage: "generating" })`
      7. Build the system prompt per §7 — see buildSystemPrompt below.
      8. `streamText({ model: gateway("anthropic/claude-haiku-4.5"), system, prompt: query, temperature: 0.3 })`. Wrap in try/catch: on error (e.g. GatewayAuthenticationError when AI_GATEWAY_API_KEY missing — RESEARCH Pitfall 1) emit `{ type: "error", code: "generation_failed", message: <err msg>, retryable: false }`, then `done`, close, return.
      9. `let fullText = ""; for await (const delta of result.textStream) { fullText += delta; emit({ type: "token", text: delta }); }`
      10. `const { citations, displayMap } = resolveCitations(fullText, sources);` (RESEARCH anti-pattern: run ONLY on the complete text)
      11. `emit({ type: "citations", citations, displayMap })`
      12. `emit({ type: "done", runId: \`run_${Date.now()}\`, usage: { inputTokens: 0, outputTokens: 0, durationMs: 0 } })`
      13. `controller.close()`
    - Return `new Response(stream, { headers: { "Content-Type": "text/event-stream", "Cache-Control": "no-cache", "Connection": "keep-alive" } })` (RESEARCH Pitfall 4 — these headers prevent buffering).

    `buildSystemPrompt(sources)` — verbatim §7 contract:
    ```
    You are answering using ONLY the sources below. Cite every factual claim
    with the source marker in square brackets, e.g. [src_3]. You may cite
    multiple sources for one claim: [src_1][src_4]. If the sources do not
    support an answer, say so — do not use outside knowledge for factual
    claims about treatments, dosing, safety, or efficacy.

    <sources>
    [src_1] (<one-line provenance label>) <source.text>
    ...
    </sources>
    ```
    For each source, the provenance label is corpus + a short title: e.g. `FDA label — BOTOX Cosmetic` for fda, `PubMed ${pmid}` for pubmed, `Practice dossier — ${doc_type}` for practice. Use `source.retrievalId` (src_N) as the marker. Instruct the model to cite ONLY src markers that appear in the list (RESEARCH Pitfall 3 — resolveCitations strips inventions, but the prompt should discourage them).

    Do NOT use `toTextStreamResponse()` (RESEARCH anti-pattern — wrong event format). Do NOT import from `lib/mock/`.
